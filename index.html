    const nombreMeses = [

      "Enero","Febrero","Marzo","Abril","Mayo","Junio",

      "Julio","Agosto","Septiembre","Octubre","Noviembre","Diciembre"

    ];

 

    let tareasRecurrentes = [];

    let festivosPersonalizados = [];

 

    // Festivos fijos México (simplificado) [web:57][web:61][web:64]

    function festivosOficialesMexico(anio, mes) {

      const fixes = [

        { mes: 1,  dia: 1 },   // Año Nuevo

        { mes: 5,  dia: 1 },   // Día del Trabajo

        { mes: 9,  dia: 16 },  // Independencia

        { mes: 12, dia: 25 }   // Navidad

      ];

      return fixes

        .filter(h => h.mes === mes)

        .map(h => h.dia);

    }

 

    function esFestivo(dateObj, festivosDias) {

      const dia = dateObj.getDate();

      return festivosDias.includes(dia);

    }

 

    function esFinDeSemana(dateObj) {

      const d = dateObj.getDay(); // 0 Dom - 6 Sáb

      return d === 0 || d === 6;

    }

 

    function generarCalendario() {

      const mes = parseInt(document.getElementById("mes").value, 10);

      const anio = parseInt(document.getElementById("anio").value, 10);

 

      if (isNaN(mes) || isNaN(anio) || mes < 1 || mes > 12) {

        alert("Mes o año inválidos.");

        return;

      }

 

      const festivosOf = festivosOficialesMexico(anio, mes);

      const festivos = Array.from(new Set([...festivosOf, ...festivosPersonalizados])).sort((a,b) => a-b);

 

      document.getElementById("fest-lista").innerHTML =

        festivos.length

          ? "Festivos este mes: " + festivos.map(d => `<span class="pill">${d}</span>`).join("")

          : "Sin festivos aún.";

 

      document.getElementById("header-mes-anio").textContent =

        `${nombreMeses[mes - 1]} ${anio}`;

 

      const tbody = document.getElementById("calendar-body");

      tbody.innerHTML = "";

 

      const firstDay = new Date(anio, mes - 1, 1);

      const lastDay = new Date(anio, mes, 0);

      const numDias = lastDay.getDate();

 

      let offset = firstDay.getDay(); // 0=Dom .. 6=Sáb

      if (offset === 0) offset = 7;   // Domingo -> 7

      let celdasVaciasInicio = (offset - 1 + 7) % 7; // 1=Lunes

 

      let diaActual = 1;

 

      // también construiremos una estructura por semanas para el Gantt

      const semanas = []; // cada semana: {inicioDia, finDia, rowElement, ganttRow}

      while (diaActual <= numDias) {

        const tr = document.createElement("tr");

        const ganttRow = document.createElement("div");

        ganttRow.className = "week-gantt";

 

        let semanaInicio = null;

        let semanaFin = null;

 

        for (let col = 0; col < 7; col++) {

          const td = document.createElement("td");

 

          if (celdasVaciasInicio > 0) {

            celdasVaciasInicio--;

          } else if (diaActual <= numDias) {

            const dateObj = new Date(anio, mes - 1, diaActual);

            const spanNum = document.createElement("div");

            spanNum.className = "day-number";

            spanNum.textContent = diaActual;

            td.appendChild(spanNum);

 

            if (esFestivo(dateObj, festivos)) {

              td.classList.add("festivo");

            }

 

            td.dataset.day = diaActual;

            if (semanaInicio === null) semanaInicio = diaActual;

            semanaFin = diaActual;

            diaActual++;

          }

 

          tr.appendChild(td);

        }

 

        tr.appendChild(ganttRow);

        tbody.appendChild(tr);

 

        if (semanaInicio !== null) {

          semanas.push({

            inicioDia: semanaInicio,

            finDia: semanaFin,

            rowElement: tr,

            ganttRow

          });

        }

      }

 

      dibujarTareasGantt(mes, anio, festivos, semanas);

    }

 

    // Calcula rangos de días hábiles para tareas recurrentes [web:63][web:66][web:69]

    function obtenerRangosRecurrentes(anio, mes, tarea, festivosDias) {

      const rangos = [];

      const inicio = new Date(anio, mes - 1, tarea.diaInicio);

      let fechaActual = new Date(inicio.getTime());

      let diasConsumidos = 0;

      let rangoActual = null;

 

      while (diasConsumidos < tarea.duracion) {

        if (!esFinDeSemana(fechaActual) && !esFestivo(fechaActual, festivosDias)) {

          const diaMes = fechaActual.getDate();

          if (!rangoActual) {

            rangoActual = { inicio: diaMes, fin: diaMes };

          } else {

            // si es el día siguiente, extendemos; si no, cerramos y empezamos otro

            if (diaMes === rangoActual.fin + 1) {

              rangoActual.fin = diaMes;

            } else {

              rangos.push(rangoActual);

              rangoActual = { inicio: diaMes, fin: diaMes };

            }

          }

          diasConsumidos++;

        }

        fechaActual.setDate(fechaActual.getDate() + 1);

        if (fechaActual.getMonth() !== mes - 1) break;

      }

      if (rangoActual) rangos.push(rangoActual);

      return rangos;

    }

 

    // Rangos para no recurrentes (días corridos)

    function obtenerRangosNoRecurrentes(anio, mes, tarea) {

      const rangos = [];

      const inicio = new Date(anio, mes - 1, tarea.diaInicio);

      let fechaActual = new Date(inicio.getTime());

      let rangoActual = null;

 

      for (let i = 0; i < tarea.duracion; i++) {

        if (fechaActual.getMonth() !== mes - 1) break;

        const diaMes = fechaActual.getDate();

        if (!rangoActual) {

          rangoActual = { inicio: diaMes, fin: diaMes };

        } else {

          if (diaMes === rangoActual.fin + 1) {

            rangoActual.fin = diaMes;

          } else {

            rangos.push(rangoActual);

            rangoActual = { inicio: diaMes, fin: diaMes };

          }

        }

        fechaActual.setDate(fechaActual.getDate() + 1);

      }

      if (rangoActual) rangos.push(rangoActual);

      return rangos;

    }

 

    function dibujarTareasGantt(mes, anio, festivosDias, semanas) {

      const addBarToWeek = (semana, inicioDia, finDia, cssClase, nombre) => {

        // columnas 1-7 dentro de la semana

        const colInicio = inicioDia - semana.inicioDia + 1;

        const colFin   = finDia   - semana.inicioDia + 1;

        if (colInicio < 1 || colFin < 1 || colInicio > 7 || colFin > 7) return;

 

        const bar = document.createElement("div");

        bar.className = `tarea-bar ${cssClase}`;

        bar.textContent = nombre;

        bar.style.gridColumn = `${colInicio} / ${colFin + 1}`;

        semana.ganttRow.appendChild(bar);

      };

 

      // Recurrentes

      tareasRecurrentes.forEach(t => {

        const rangos = obtenerRangosRecurrentes(anio, mes, t, festivosDias);

        rangos.forEach(r => {

          semanas.forEach(semana => {

            const inicio = Math.max(r.inicio, semana.inicioDia);

            const fin    = Math.min(r.fin, semana.finDia);

            if (inicio <= fin) {

              addBarToWeek(semana, inicio, fin, "tarea-recurrente", t.nombre);

            }

          });

        });

      });

 

      // No recurrentes

      tareasNoRecurrentes.forEach(t => {

        const rangos = obtenerRangosNoRecurrentes(anio, mes, t);

        rangos.forEach(r => {

          semanas.forEach(semana => {

            const inicio = Math.max(r.inicio, semana.inicioDia);

            const fin    = Math.min(r.fin, semana.finDia);

            if (inicio <= fin) {

              addBarToWeek(semana, inicio, fin, "tarea-norec", t.nombre);

            }

          });

        });

      });

    }

      renderListaRecurrentes();

      generarCalendario();

    });

      const cont = document.getElementById("rec-lista");

      if (!tareasRecurrentes.length) {

        cont.innerHTML = "<em>Sin tareas recurrentes.</em>";

        return;

      }

      cont.innerHTML = tareasRecurrentes.map((t, idx) =>

        `<div class="pill">${idx+1}. ${t.nombre} (desde día ${t.diaInicio}, ${t.duracion} días hábiles)</div>`

      ).join("");

    }
